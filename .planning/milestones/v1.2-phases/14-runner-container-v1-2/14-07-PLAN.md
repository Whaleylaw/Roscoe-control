```typescript
export interface PreambleInput {
  isResuming: boolean
  taskId: number | string
  apiBase: string           // 'http://host.docker.internal:3000'
  priorAttempts: Array<{ started_at: number; exit_code: number | null; failure_reason: string | null }>
}
export function generatePreamble(input: PreambleInput): string  // returns Markdown body

// src/lib/runner-worktree.ts
export interface McTaskJson {
  task_id: string
  recipe_slug: string
  attempt: number
  is_resuming: boolean
  prior_attempts: Array<{ started_at: string; exit_code: number | null; failure_reason: string | null }>
}
export function seedMcDir(worktreePath: string, input: { task: McTaskJson }): void
export function readMcTaskJson(worktreePath: string): McTaskJson | null
export function writeMcTaskJson(worktreePath: string, task: McTaskJson): void
export function buildPriorAttemptsEntry(startedAtUnix: number, exitCode: number | null, reason: string | null): McTaskJson['prior_attempts'][number]

// src/lib/runner-docker.ts
export interface DockerRunInput {
  image: string
  taskId: number | string
  attempt: number
  recipeSlug: string
  runnerId: string
  runnerStartedAtIso: string
  containerName: string                 // 'mc-task-42-a3'
  worktreePath: string                  // host abs path -> /workspace:rw
  recipeStagePath: string               // host abs path -> /recipe:ro
  readOnlyMounts: Array<{ host_path: string; container_path?: string; label: string }>
  extraSkills: string[]                 // host abs paths
  envFilePath: string
  memory: string                        // '2g'
  cpus: number                          // 1.0
  networkHostGateway?: boolean          // defaults true
}
export function buildDockerRunArgs(input: DockerRunInput): string[]  // argv for spawn('docker', argv)

export interface StageRecipeInput {
  sourceDir: string                     // recipes/<slug>/
  stageDir: string                      // .data/runner/recipe-stage/task-<id>/
  preambleContents: string              // the generated preamble text
}
export function stageRecipe(input: StageRecipeInput): Promise<void>   // deep-copy sourceDir -> stageDir, then write PREAMBLE.md

export interface EnvFileInput {
  envMap: Record<string, string>
  filePath: string                      // abs path, e.g. /tmp/mc-task-42-a3.env
}
export function writeEnvFile(input: EnvFileInput): void    // writes KEY=VALUE per line, mode 0600; newline-escapes values containing \n (shouldn't but defensive)
export function cleanupEnvFile(filePath: string): void      // fs.rmSync; ignores ENOENT
```
