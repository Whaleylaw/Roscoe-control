    **Part A — inventory endpoint (src/app/api/runner/inventory/route.ts, NEW):**

    ```ts
    import { NextRequest, NextResponse } from 'next/server'
    import { requireRole } from '@/lib/auth'
    import { getDatabase } from '@/lib/db'

    /**
     * GET /api/runner/inventory
     *
     * Returns the freshest runner heartbeat's tracked task IDs. Read-only observability
     * wrapper over runner_heartbeats.metadata_json.active_task_ids.
     *
     * Phase 15 SCHED-03 / Plan 15-02: requeueStaleTasks reads the same source directly
     * from the DB. This endpoint is a thin wrapper for operators and Phase 16 UI.
     *
     * Auth: runner-secret principal (id=-1000). Runner-tokens are task-scoped and
     * MUST NOT reach this endpoint; 403 if the wrong principal presents.
     */
    export async function GET(request: NextRequest): Promise<NextResponse> {
      const auth = requireRole(request, 'operator')
      if ('error' in auth) return NextResponse.json({ error: auth.error }, { status: auth.status })

      if (auth.user.id !== -1000) {
        return NextResponse.json({ error: 'runner-secret principal required' }, { status: 403 })
      }

      const db = getDatabase()
      const nowUnix = Math.floor(Date.now() / 1000)
      const STALE_WINDOW_SECS = 90 // Same LOCKED window as reconcileRunnerHeartbeat

      const row = db.prepare(
        `SELECT runner_id, last_heartbeat_at, metadata_json
         FROM runner_heartbeats
         WHERE last_heartbeat_at >= ?
         ORDER BY last_heartbeat_at DESC
         LIMIT 1`
      ).get(nowUnix - STALE_WINDOW_SECS) as { runner_id: string; last_heartbeat_at: number; metadata_json: string | null } | undefined

      if (!row) {
        return NextResponse.json({
          runner_id: null,
          last_heartbeat_at: null,
          active_task_ids: [],
          stale: true,
        }, { status: 200 })
      }

      let activeTaskIds: number[] = []
      try {
        const parsed = JSON.parse(row.metadata_json ?? '{}')
        if (Array.isArray(parsed?.active_task_ids)) {
          activeTaskIds = parsed.active_task_ids.filter((n: unknown) => typeof n === 'number' && Number.isFinite(n) && n > 0)
        }
      } catch { /* non-fatal parse error — treat as empty */ }

      return NextResponse.json({
        runner_id: row.runner_id,
        last_heartbeat_at: row.last_heartbeat_at,
        active_task_ids: activeTaskIds,
        stale: false,
      }, { status: 200 })
    }
    ```
