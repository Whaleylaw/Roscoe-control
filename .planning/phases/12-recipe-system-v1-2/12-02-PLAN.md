<objective>
The indexer module is the single entry point every caller (boot scanner, chokidar watcher, POST /api/recipes) uses to materialise a filesystem recipe directory into a `recipes` row. It parses recipe.yaml via the Zod schema from 12-01, walks the directory to compute `dir_sha`, reads SOUL.md, skips if `dir_sha` matches the existing row, and otherwise UPSERTs — OR writes a minimal `{ slug, error_message }` row when validation fails. Removals cascade to the FTS table via the migration 059 triggers.

Purpose: Centralising the write path here means the watcher (12-03) and the POST endpoint (12-04) don't need to repeat YAML parse, model validation, or SHA computation logic. The `error_message` column flow (per CONTEXT.md "hard fail cases") is implemented here once. This is the module MODEL-02 ultimately flows through.

Output: Two TypeScript modules (`recipe-indexer.ts`, `recipe-hash.ts`), two test suites, zero API routes or watcher code.

```typescript
import { createHash } from 'crypto'
import { readFile, readdir, stat } from 'fs/promises'
import { join, relative, sep } from 'path'

/**
 * Files at the recipe root that contribute to dir_sha.
 * Order-invariant: we sort the discovered path list before hashing.
 */
const ROOT_FILES = ['recipe.yaml', 'SOUL.md', 'README.md']

/**
 * Subdirectories recursively walked for dir_sha.
 * Anything outside these (temp files, editor swap files, node_modules) is ignored.
 */
const WALKED_DIRS = ['tools', 'skills']

/**
 * Compute SHA-256 over the recipe directory's contributing contents.
 *
 * Algorithm (deterministic across platforms):
 *   1. Discover contributing files:
 *      - recipe.yaml, SOUL.md, README.md at the root (each optional — recipe.yaml
 *        being missing is the indexer's hard-fail case, NOT the hasher's)
 *      - Every file recursively under tools/ and skills/ (no file extension filter)
 *   2. Normalise each path to POSIX-style (forward slashes) relative to absDir
 *   3. Sort the path list lexicographically (UTF-8 byte order, en_US_POSIX semantics)
 *   4. For each sorted path, append to the hash:
 *        '\u0000' + relPath + '\u0000' + <fileSize:LE 64-bit> + <file bytes>
 *      Null-byte delimiters prevent path/content collisions.
 *   5. Return hex-encoded digest.
 *
 * Missing paths silently contribute nothing (recipe.yaml being missing is handled
 * by the indexer, not here — it simply results in a different hash than a recipe
 * that has one, which is fine for dedup).
 */
export async function computeDirSha(absDir: string): Promise<string> {
  const hasher = createHash('sha256')
  const files: string[] = []

  // Root files
  for (const name of ROOT_FILES) {
    const p = join(absDir, name)
    if (await exists(p) && (await stat(p)).isFile()) files.push(p)
  }

  // Walked subtrees
  for (const sub of WALKED_DIRS) {
    const subAbs = join(absDir, sub)
    if (await exists(subAbs) && (await stat(subAbs)).isDirectory()) {
      await collectFiles(subAbs, files)
    }
  }

  // Sort by POSIX-relative path for determinism
  const pairs = files.map((f) => ({
    rel: relative(absDir, f).split(sep).join('/'),
    abs: f,
  }))
  pairs.sort((a, b) => (a.rel < b.rel ? -1 : a.rel > b.rel ? 1 : 0))

  for (const { rel, abs } of pairs) {
    const bytes = await readFile(abs)
    const sizeBuf = Buffer.alloc(8)
    sizeBuf.writeBigUInt64LE(BigInt(bytes.length))
    hasher.update(Buffer.from([0])) // null delimiter
    hasher.update(Buffer.from(rel, 'utf8'))
    hasher.update(Buffer.from([0]))
    hasher.update(sizeBuf)
    hasher.update(bytes)
  }

  return hasher.digest('hex')
}

async function collectFiles(dir: string, out: string[]): Promise<void> {
  const entries = await readdir(dir, { withFileTypes: true })
  for (const entry of entries) {
    const abs = join(dir, entry.name)
    if (entry.isDirectory()) {
      await collectFiles(abs, out)
    } else if (entry.isFile()) {
      out.push(abs)
    }
    // Symlinks and special files are skipped (not counted in dir_sha).
  }
}

async function exists(p: string): Promise<boolean> {
  try {
    await stat(p)
    return true
  } catch {
    return false
  }
}
```

```typescript
  })
```

```typescript
import { readFile } from 'fs/promises'
import { join, basename } from 'path'
import type Database from 'better-sqlite3'
import { getDatabase } from './db'
import { logger } from './logger'
import { parseRecipeYaml } from './recipe-schema'
import { computeDirSha } from './recipe-hash'

/**
 * Outcome of an indexRecipe call.
 * - 'indexed'  — row inserted/updated with a valid recipe
 * - 'unchanged' — dir_sha matched existing row; DB not touched
 * - 'error'    — validation failed; minimal error-row written with error_message
 * - 'skipped_missing' — recipe.yaml does not exist in absDir; NO row written
 *                       (caller should call removeRecipe if a row exists)
 */
export type IndexResult =
  | { status: 'indexed'; slug: string; dirSha: string }
  | { status: 'unchanged'; slug: string; dirSha: string }
  | { status: 'error'; slug: string; error: string }
  | { status: 'skipped_missing'; slug: string }

export interface IndexOptions {
  /** workspace_id to scope the row. Defaults to 1 (global). */
  workspaceId?: number
  /** tenant_id to scope the row. Defaults to 1. */
  tenantId?: number
  /** Force reindex even if dir_sha matches (used by POST /api/recipes and resync). */
  force?: boolean
  /** Override getDatabase() — for tests. */
  dbOverride?: Database.Database
}

/**
 * Index a recipe directory into the DB.
 *
 * Flow:
 *   1. Derive slug from basename(absDir)
 *   2. If recipe.yaml does not exist → 'skipped_missing' (caller decides whether to removeRecipe)
 *   3. Compute dir_sha
 *   4. If existing row has same dir_sha AND !force → 'unchanged' (fast path)
 *   5. Read recipe.yaml, parseRecipeYaml
 *   6. On parse failure → UPSERT minimal error row; log.error; return 'error'
 *      (ALSO: if recipe.yaml's slug field disagrees with basename(absDir), that's a
 *       parse failure with error "slug mismatch: directory is 'X' but recipe.yaml says 'Y'")
 *   7. Read SOUL.md (optional; null if missing)
 *   8. UPSERT full row with all JSON fields serialised; return 'indexed'
 */
export async function indexRecipe(absDir: string, opts: IndexOptions = {}): Promise<IndexResult> {
  const workspaceId = opts.workspaceId ?? 1
  const tenantId = opts.tenantId ?? 1
  const db = opts.dbOverride ?? getDatabase()
  const slug = basename(absDir)

  // Step 2: recipe.yaml existence check
  let recipeYamlRaw: string
  try {
    recipeYamlRaw = await readFile(join(absDir, 'recipe.yaml'), 'utf8')
  } catch {
    return { status: 'skipped_missing', slug }
  }

  // Step 3: dir_sha
  const dirSha = await computeDirSha(absDir)

  // Step 4: fast-path dedup
  const existing = db.prepare(`SELECT dir_sha, error_message FROM recipes WHERE slug = ?`).get(slug) as
    | { dir_sha: string; error_message: string | null }
    | undefined
  if (!opts.force && existing && existing.dir_sha === dirSha && existing.error_message === null) {
    return { status: 'unchanged', slug, dirSha }
  }

  // Step 5: parse
  const parsed = parseRecipeYaml(recipeYamlRaw)
  if (!parsed.ok) {
    logger.error({ slug, path: absDir, reason: parsed.error }, 'recipe index failed')
    writeErrorRow(db, slug, parsed.error, workspaceId, tenantId)
    return { status: 'error', slug, error: parsed.error }
  }

  // Step 6: slug-vs-directory consistency check
  if (parsed.value.slug !== slug) {
    const reason = `slug mismatch: directory is '${slug}' but recipe.yaml says '${parsed.value.slug}'`
    logger.error({ slug, path: absDir, reason }, 'recipe index failed')
    writeErrorRow(db, slug, reason, workspaceId, tenantId)
    return { status: 'error', slug, error: reason }
  }

  // Step 7: optional SOUL.md
  let soulMd: string | null = null
  try {
    soulMd = await readFile(join(absDir, 'SOUL.md'), 'utf8')
  } catch {
    soulMd = null
  }

  // Step 8: UPSERT the full row. INSERT OR REPLACE guarantees FTS triggers fire with
  // fresh values (trigger emits 'delete' + 'insert' on UPDATE — see migration 059).
  writeIndexedRow(db, parsed.value, dirSha, soulMd, workspaceId, tenantId)
  return { status: 'indexed', slug, dirSha }
}

/**
 * Delete a recipe row by slug. The migration 059 AFTER DELETE trigger cascades to
 * recipes_fts. No-op if the row does not exist.
 */
export function removeRecipe(slug: string, opts: { dbOverride?: Database.Database } = {}): { removed: boolean } {
  const db = opts.dbOverride ?? getDatabase()
  const result = db.prepare(`DELETE FROM recipes WHERE slug = ?`).run(slug)
  return { removed: result.changes > 0 }
}

/**
 * Read an indexed recipe by slug, parsing JSON columns back into their declared
 * types. Used by the GET /api/recipes/:slug handler and the recipe dispatch path.
 *
 * Returns:
 *   - RecipeRow when error_message IS NULL (fully indexed)
 *   - RecipeErrorRow when error_message IS NOT NULL (broken recipe)
 *   - null when no row exists
 */
export function getIndexedRecipeBySlug(
  slug: string,
  opts: { dbOverride?: Database.Database } = {}
): import('../types/recipe').RecipeRow | import('../types/recipe').RecipeErrorRow | null {
  const db = opts.dbOverride ?? getDatabase()
  const row = db.prepare(`SELECT * FROM recipes WHERE slug = ?`).get(slug) as
    | Record<string, unknown>
    | undefined
  if (!row) return null

  if (row.error_message) {
    return {
      id: row.id as number,
      slug: row.slug as string,
      error_message: row.error_message as string,
      created_at: row.created_at as number,
      updated_at: row.updated_at as number,
    }
  }

  return {
    id: row.id as number,
    slug: row.slug as string,
    name: row.name as string,
    description: (row.description as string) ?? null,
    when_to_use: (row.when_to_use as string) ?? null,
    image: row.image as string,
    workspace_mode: row.workspace_mode as 'worktree' | 'readonly' | 'none',
    timeout_seconds: row.timeout_seconds as number,
    max_concurrent: row.max_concurrent as number,
    env: JSON.parse((row.env_json as string) || '{}'),
    secrets: JSON.parse((row.secrets_json as string) || '[]'),
    tags: JSON.parse((row.tags_json as string) || '[]'),
    model: JSON.parse((row.model_json as string) || '{}'),
    version: row.version as number,
    dir_sha: row.dir_sha as string,
    soul_md: (row.soul_md as string) ?? null,
    error_message: null,
    workspace_id: row.workspace_id as number,
    tenant_id: row.tenant_id as number,
    created_at: row.created_at as number,
    updated_at: row.updated_at as number,
  }
}

    JSON.stringify(y.model),

function writeErrorRow(
  db: Database.Database,
  slug: string,
  errorMessage: string,
  workspaceId: number,
  tenantId: number,
): void {
  // Error rows carry minimal data: slug, error_message, scoping, timestamps.
  // We still need non-null values for columns that are NOT NULL in the schema —
  // use zero-ish defaults so the row round-trips through PRAGMA without errors.
  db.prepare(`
    INSERT INTO recipes
      (slug, name, image, workspace_mode, timeout_seconds, dir_sha,
       env_json, secrets_json, tags_json, model_json, max_concurrent, version,
       error_message, workspace_id, tenant_id, updated_at)
    VALUES (?, ?, ?, ?, ?, ?, '{}', '[]', '[]', '{}', 1, 1, ?, ?, ?, unixepoch())
    ON CONFLICT(slug) DO UPDATE SET
      error_message=excluded.error_message,
      updated_at=unixepoch()
  `).run(
    slug,
    // Placeholders for NOT NULL columns when we don't have a valid parse
    slug,              // name -> fallback to slug
    'unknown',         // image placeholder
    'worktree',        // workspace_mode placeholder
    0,                 // timeout_seconds placeholder
    '',                // dir_sha empty for error rows
    errorMessage,
    workspaceId,
    tenantId,
  )
}
```

```typescript
  it('removeRecipe deletes the row and cascade-clears FTS', async () => {

  it('transitioning from error row to valid flips error_message back to NULL', async () => {
    writeFileSync(join(recipeDir, 'recipe.yaml'), 'not valid yaml :: [')

    const fixed = await indexRecipe(recipeDir, { dbOverride: db })
    expect(fixed.status).toBe('indexed')
    const healthy = getIndexedRecipeBySlug('hello', { dbOverride: db })
    expect((healthy as any).error_message).toBeNull()
    expect((healthy as any).name).toBe('Hello')
  })
```
