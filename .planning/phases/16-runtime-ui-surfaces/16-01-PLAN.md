**Step 8 — Create `src/app/api/runtime/runner-status/route.ts`:**
Viewer-auth GET endpoint. Follow the project's standard route pattern from `CONVENTIONS.md` § Error Handling:
```typescript
import { NextRequest, NextResponse } from 'next/server'
import { requireRole } from '@/lib/auth'
import { getDatabase } from '@/lib/db'
import { logger } from '@/lib/logger'

const STALE_WINDOW_SECS = 90 // matches task-dispatch.ts + inventory/route.ts per Plan 15-06 LOCK

export async function GET(request: NextRequest): Promise<NextResponse> {
  const auth = requireRole(request, 'viewer')
  if ('error' in auth) return NextResponse.json({ error: auth.error }, { status: auth.status })

  try {
    const db = getDatabase()
    const nowUnix = Math.floor(Date.now() / 1000)

    const heartbeatRow = db.prepare(
      `SELECT last_heartbeat_at FROM runner_heartbeats
       WHERE last_heartbeat_at >= ?
       ORDER BY last_heartbeat_at DESC LIMIT 1`
    ).get(nowUnix - STALE_WINDOW_SECS) as { last_heartbeat_at: number } | undefined

    // Workspace-scoped waiting count per 16-CONTEXT.md + research Open Question 3.
    // auth.user.workspace_id is the authenticated user's workspace.
    const waitingRow = db.prepare(
      `SELECT COUNT(*) AS n FROM tasks
       WHERE workspace_id = ?
         AND recipe_slug IS NOT NULL
         AND status IN ('inbox', 'assigned')`
    ).get(auth.user.workspace_id) as { n: number }

    return NextResponse.json({
      online: heartbeatRow != null,
      last_heartbeat_at: heartbeatRow?.last_heartbeat_at ?? null,
      tasks_waiting: waitingRow.n ?? 0,
    }, { status: 200 })
  } catch (error) {
    logger.error({ err: error }, 'GET /api/runtime/runner-status error')
    return NextResponse.json({ error: 'Failed to read runner status' }, { status: 500 })
  }
}
```
IMPORTANT — verify `auth.user.workspace_id` is the correct property name by grepping `requireRole` / existing viewer-auth routes (`src/app/api/status/route.ts` is the precedent per research). If the field is named differently (e.g. `tenantId`), use that instead. Do NOT hand-roll auth — reuse `requireRole`.
