    **Test body (the cold-start journey):**

    ```typescript
    test.describe('Phase 8: Projects entry point — cold-start journey (NAV-01)', () => {
      let projectId: number
      const slug = 'e2e-phase-8'

      test.beforeAll(async ({ request }) => {
        const res = await request.post('/api/projects', {
          headers: API_KEY_HEADER,
          data: { name: 'E2E Phase 8', ticket_prefix: 'E8P', slug },
        })
        expect(res.ok()).toBeTruthy()
        const body = await res.json()
        projectId = body.project.id
      })

      test.afterAll(async ({ request }) => {
        if (projectId) {
          await request.delete(`/api/projects/${projectId}`, { headers: API_KEY_HEADER }).catch(() => {})
        }
      })

      test('user reaches project workspace via nav-rail and returns via breadcrumb', async ({ page }) => {
        // 1. Start at root (logged-in state assumed via storageState or explicit login step)
        await page.goto('/')

        // 2. Click the Projects nav-rail item
        await page.getByRole('button', { name: /^projects$/i }).first().click()
        await page.waitForURL('**/projects')
        await expect(page).toHaveURL(/\/projects$/)

        // 3. Click the project row for the test fixture
        await page.getByRole('button', { name: /E2E Phase 8/ }).click()
        await page.waitForURL(`**/project/${slug}`)
        await expect(page).toHaveURL(new RegExp(`/project/${slug}`))

        // 4. Confirm the workspace dashboard is rendered (look for a dashboard-identifying element)
        // The dashboard has a project brief / status overview — assert one of those is visible.
        // Use any stable text from the dashboard; if uncertain, wait for the breadcrumb itself:
        await expect(page.getByRole('navigation', { name: /breadcrumb/i })).toBeVisible()

        // 5. Click the breadcrumb 'Projects' segment
        await page.getByRole('navigation', { name: /breadcrumb/i })
          .getByRole('button', { name: /projects/i })
          .first()
          .click()

        // 6. Verify we landed back at /projects (NOT /)
        await page.waitForURL('**/projects')
        await expect(page).toHaveURL(/\/projects$/)
        await expect(page).not.toHaveURL(/^http[s]?:\/\/[^/]+\/$/)
      })
    })
    ```

    **Notes on selector choices:**

    - `getByRole('button', { name: /^projects$/i }).first()` targets the nav-rail Projects item. The `.first()` disambiguates if multiple nav items happen to translate to "Projects" in the current locale; the top-most (nav-rail) one is what we want.
    - `getByRole('button', { name: /E2E Phase 8/ })` targets the project row in the panel — it's a full-row clickable button per Plan 08-01's implementation (role="button" with aria-label=project.name).
    - `getByRole('navigation', { name: /breadcrumb/i })` targets the breadcrumb component (the component already sets `aria-label="Breadcrumb"` on line 36 of project-breadcrumb.tsx).
